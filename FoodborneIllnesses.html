# draw the choloropleth map
import plotly.plotly as py
import numpy as np

for col in dff.columns:
    dff[col] = dff[col].astype(str)

group_state = dff.groupby('State',sort=False).Illnesses.count().reset_index()

dff['text'] = dff['State'] + '<br>' +\
    'Illness '+dff['Illnesses']
    
scl = [[0.0, 'rgb(149,249,67)'],[0.2, 'rgb(204,249,67)'],[0.4, 'rgb(237,249,67)'],\
            [0.6, 'rgb(249,227,67)'],[0.8, 'rgb(249,270,67)'],[1.0, 'rgb(249,116,67)']]
s = group_state['State']
data = [ dict(
        type='choropleth',
        colorscale = scl,
        autocolorscale = False,
        locations = s,
        z = group_state['Illnesses'],
        locationmode = 'USA-states',
        marker = dict(
            line = dict (
                color = 'rgb(255,255,255)',
                width = 2
            ) ),
        colorbar = dict(
            title = "Illnesses Count")
        ) ]

layout = dict(
        title = 'Foodborne Illnesses 1998-2015<br>(Hover for breakdown)',
        geo = dict(
            scope='usa',
            projection=dict( type='albers usa' ),
            showlakes = True,
            lakecolor = 'rgb(255, 255, 255)'),
             )
    
fig = dict( data=data, layout=layout )
py.iplot( fig, filename='d3-cloropleth-map' )

from sklearn.naive_bayes import GaussianNB
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction import DictVectorizer
from sklearn.metrics import accuracy_score


cols_to_dict = [ 'Year', 'Month', 'State', 'Location', 'Food','Species' ]
category_dict = dff[ cols_to_dict ].to_dict( orient = 'records' )


dff.drop('Ingredient',axis=1,inplace=True)
dff.drop('Serotype/Genotype',axis=1,inplace=True)

dff['Species'].fillna('Norovirus genogroup I',inplace=True)
dff['Location'].fillna('Restaurant',inplace=True)
dff['Food'].fillna('Multiple Foods',inplace=True)
dff['Hospitalizations'].fillna('1',inplace=True)
dff['Fatalities'].fillna('1',inplace=True)

#features_train,features_test,labels_train,labels_test = train_test_split(dff[features],dff[labels], test_size=0.3, random_state=42)
v = DictVectorizer(sparse=False)
X = v.fit_transform(category_dict)
nb_clf = GaussianNB()
features_train,features_test,labels_train,labels_test = train_test_split(X,dff['Status'], test_size=0.3, random_state=42)

nb_clf.fit(features_train,labels_train)
p = nb_clf.predict(features_test)
print 'Accuracy = ',accuracy_score(p,labels_test)
